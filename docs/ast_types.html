XDC parser AST types

compile_unit_t
  • import_t[] imports
  • function_t[] functions
import_t
  • string module_name
function_t
  • string name
  • data_type_t return_type
  • variable_declaration_t[] parameters
  • code_block_t body
variable_declaration_t
  • data_type_t type
  • string name
variable_definition_t
  • data_type_t type
  • string name
  • expression_t initializer
data_type_t
  • data_type_type_t type
  • ( basic_type_t | static_array_type_t | dynamic_array_type_t )
basic_type_t
  • ( boolean_t | int_t )
static_array_type_t
  • basic_type_t data_type
  • integer length
code_block_t
  • statment_t[] statment
statment_t
  • statment_type_t type
  • ( variable_definition_t | assigment_t | code_block_t | for_loop_t | if_else_t | function_call_t | return_t )
if_else_t
  • if_statment_t[] if_else_blocks
  • code_block_t else_block
if_statment_t
  • expression_t condition
  • code_block_t body
assigment_t
  • variable_ref_t target
  • expression_t value
expression_t
  • expression_type_t type
  • ( unary_operation_t | binary_operation_t | constant_t | variable_ref_t | function_call_t )
unary_operation_t
  • unary_operation_type_t operation
  • expression_t operand
binary_operation_t
  • binary_operation_type_t operation
  • expression_t left
  • expression_t right
constant_t
  • basic_type_t type
  • ( integer | bool | float )
return_t
  • expression_t return_value
variable_ref_t
  • variable_ref_type_t type
  • ( string | array_cell_ref_t)
for_loop_t
  • ? init
  • expression_t test
  • ? step
  • code_block_t body
function_call_t
  • string name
  • expression_t[] arguments
array_cell_ref_t
  • string name
  • integer index
